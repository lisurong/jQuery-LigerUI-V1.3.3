addRowButtonClick参数 示例：

 1 var dialog = null, combobox;
 2         $(function ()
 3         { 
 4             combobox = $("#test1").ligerComboBox({
 5                 data: [
 6                     { text: '张三', id: '1' },
 7                     { text: '李四', id: '2' },
 8                     { text: '赵武', id: '44' }
 9                 ],
10                 emptyText: '（空）',            //空行
11                 addRowButton: '新增',           //新增按钮
12                 addRowButtonClick: function ()  //新增事件
13                 {
14                     combobox.clear();
15                     dialog = $.ligerDialog.open({
16                         height: 300,
17                         width: 400,
18                         title: '打开窗口提示',
19                         url: 'addNew.htm',
20                         showMax: false,
21                         showToggle: true,
22                         showMin: false,
23                         isResize: true,
24                         slide: false,
25                         data: {
26                             callback: afterSave
27                         }
28                     });
29                 }
30             });
31         });
32         function afterSave(data)
33         {
34             dialog.close(); 
35             combobox.addItem(data); 
36             combobox.setValue(data.id, data.text);
37         }
