ligerMenuBar

菜单条组件

示例

 1  var menu1 = { width: 120, items:
 2             [
 3             { text: '保存', click: itemclick },
 4             { text: '列存为', click: itemclick },
 5             { line: true },
 6             { text: '关闭', click: itemclick }
 7             ]
 8         };
 9 
10         var menu2 = { width: 120, items:
11             [
12             {
13                 text: '文件', children:
14                 [
15                     { text: 'Excel', click: itemclick },
16                     { text: 'Word', click: itemclick },
17                     { text: 'PDF', click: itemclick },
18                     { text: 'TXT', click: itemclick },
19                     { line: true },
20                     { text: 'XML', click: itemclick }
21                 ]
22             },
23             ]
24         };
25         function itemclick(item)
26         {
27             alert(item.text);
28         }
29         $(function ()
30         {
31             $("#topmenu").ligerMenuBar({ items: [
32                 { text: '文件', menu: menu1 },
33                 { text: '导出', menu: menu2 }
34             ]
35             });
36 
37             $("#toptoolbar").ligerToolBar({ items: [
38                 { text: '增加', click: itemclick , icon:'add'},
39                 { line:true },
40                 { text: '修改', click: itemclick },
41                 { line:true },
42                 { text: '删除', click: itemclick }
43             ]
44             });
45         });
