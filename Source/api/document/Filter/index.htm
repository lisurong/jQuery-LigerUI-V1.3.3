ligerFilter

过滤器（自定义查询组件）

示例

 1  var fields = [
 2                 { display: '主键', name: 'CustomerID' },
 3                 { display: '公司名', name: 'CompanyName', editor: { type: 'string'} },
 4                 { display: '加入时间', name: 'AddTime', editor: { type: 'date', options: { showTime: true}} },
 5                 { display: '金额', name: 'Amount', type: 'number', editor: { type: 'number', options: { type: 'float'}} }
 6             ];
 7 
 8 
 9             var filter = $("#filter").ligerFilter({ fields: fields });
10 
11             $("#Button1").click(function ()
12             {
13                 var group = filter.getData();
14                 $("#txtGroup").val($.ligerui.toJSON(group));
15             });
