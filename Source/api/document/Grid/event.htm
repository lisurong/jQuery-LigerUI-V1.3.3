42         function f_onBeforeEdit(e)

48         function f_onBeforeSubmitEdit(e)
49         { 
50             if (e.columnname == "Age")
51             {
52                 if (e.value < 20 || e.value > 30) return false;
53             }
54             return true;
55         }
56         //编辑后事件

58         {
59             if (e.columnname == "Age")
60             {
61                 alert(e.value);
62             }
63         }
